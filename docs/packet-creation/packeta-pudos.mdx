# Packeta PUDOs

When creating packets to our Packeta PUDOs through our API you have to select the destination PUDO using
it's ID obtained from our [`Widget`](../pudo-delivery/widget) or from our [Packeta PUDOs feed](../pudo-delivery/packeta-pudos).

<TabItem value="php" label="PHP">

```php

<?php

$apiPassword = 'apiPassword';

$packetAttributes = [
    'number' => '20220513001',
    'name' => 'John',
    'surname' => 'Doe',
    'company' => 'Company',
    'email' => 'example@packetatest.com',
    'phone' => '+420777123456',
    'addressId' => 52,
    'cod' => 115,
    'value' => '115',
    'currency' => 'CZK',
    'weight' => 2.5,
    'eshop' => 'MyEshop',
];

try {
    $client = new SoapClient('https://www.zasilkovna.cz/api/soap.wsdl');
    $result = $client->createPacket($apiPassword, $packetAttributes);

    var_dump($result);
} catch (SoapFault $e) {
    echo $e->getMessage();

    if (isset($e->detail->PacketAttributesFault)) {
        var_dump($e->detail->PacketAttributesFault->attributes->fault);
    } elseif (isset($e->detail->IncorrectApiPasswordFault)) {
        var_dump('Incorrect API password.');
    }
}
?>
```

<TabItem value="csharp" label="C#">

```csharp

using System;
using System.ServiceModel;
using System.Threading.Tasks;
using PacketaService;

namespace PacketaClientExample
{
    class Program
    {
        static async Task Main(string[] args)
        {
            BasicHttpsBinding binding = new BasicHttpsBinding();
            EndpointAddress endpointAddress = new EndpointAddress("https://www.zasilkovna.cz/api/soap");

            const string apiPassword = "apiPassword";
            PacketAttributes packetAttributes = new PacketAttributes
            {
                number = "orderNumber"
                name = "John",
                surname = "Doe",
                company = "Company",
                email = "example@packetatest.com",
                phone = "+420777777777",
                addressId = 1817,
                cod = 115,
                value = 115,
                currency = "CZK",
                weight = 2.5,
                eshop = "MyEshop",

            };

            PacketaClient client = new PacketaClient(binding, endpointAddress);
            try
            {
                createPacketResponse response = await client.createPacketAsync(new createPacketRequest(apiPassword, packetAttributes));

                //Process response
                Console.WriteLine(response.createPacketResult.id);
            }
            catch(Exception e)
            {
                Console.WriteLine(e.Message);
            }      
        }
    }
}


```

<TabItem value="js" label="Node.js">

```js

import { Builder, Parser } from "xml2js"

const requestBody = {
    createPacket: {
        apiPassword: "apiPassword",
        packetAttributes: {
            number: "orderNumber",
            name: "John",
            surname: "Doe",
            company: "Company",
            email: "example@packetatest.com",
            phone: "+420777777777",
            addressId: 1817,
            cod: 115,
            value: 115,
            currency: EUR,
            weight: 2.5,
            eshop: "MyEshop",

        }
    }
}

try {
    const response = await fetch(
        "https://www.zasilkovna.cz/api/rest",
        {
            method: "POST",
            body: new Builder().buildObject(requestBody)
        }
    );

    const responseBody = await new Parser({ explicitArray: false }).parseStringPromise(await response.text());

    //process response
    console.log(responseBody.response.result.id);
}
catch (exception) {
    console.log(exception);
}


```
